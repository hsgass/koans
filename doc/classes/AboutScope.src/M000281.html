  def test_you_can_reference_nested_classes_using_the_scope_operator
    fido = Jims::Dog.new
    rover = Joes::Dog.new
    assert_equal __, fido.identify
    assert_equal __, rover.identify

    assert_equal __, fido.class != rover.class
    assert_equal __, Jims::Dog != Joes::Dog
  end
